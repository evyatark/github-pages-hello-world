סערה ברחוב דרך בית לחם

סְעָרָה בִּרְחוֹב דֶּרֶךְ בֵּית לֶחֶם הוֹלֶכֶת וּמִתְחַזֶּקֶת, הָעֵצִים הַצְּעִירִים מִטַּלְטְלִים, נִגּוּן עַתִּיק מְנַעֵר אוֹתָם לַצְּדָדִים, הַשָּׁמַיִם אֲפֹרִים כְּחֻלִּים אֲפֹרִים, הַיְּרֻקִּים יְרֻקִּים בְּכַמָּה יְרֻקִּים, וְהַכִּנּוֹרוֹת עוֹנִים בְּקַוִּים אֲרֻכִּים, בְּמֵיתָרִים שֶׁסּוֹטְרִים לָאֲוִיר עֲצַבֵּיהֶם רוֹעֲדִים. סוֹדְקִים בָּאֲוִיר הַגְּשָׁמִים, הוֹפְכִים אוֹתוֹ נִרְאֶה. טִפָּה נוֹפֶלֶת מִתְגַּבֶּרֶת עַל כָּל מֶרְחָק בָּעוֹלָם. זֶה הַשָּׂדֶה שֶׁאֲנִי חוֹצָה בַּשָּׁעוֹת הַנִּבְדָּלוֹת כְּלֵי הַקֶּשֶׁת דּוֹלְפִים רְגָשׁוֹת מֵהַצְּלָעוֹת, הָעֵשֶׂב עַד הַבִּרְכַּיִם, מִכְפָּלוֹת הַשִּׂמְלָה נִסְחָבוֹת, הָרוּחוֹת מִתְיַפְּחוֹת לְיַבֵּשׁ דְּמָעוֹת, הַחֲבֵרוֹת צוֹחֲקוֹת בַּבַּיִת בְּצֶלַע הָהָר וַאֲנִי בַּדֶּרֶךְ, כֻּלִּי בַּדֶּרֶךְ, אֲנִי הַדֶּרֶךְ. צְעָדִים מְתוּנִים, מוֹדֵרָטוֹ מַחְשָׁבוֹת. הַתְּלָאוֹת: רֹחַב הַשָּׂדֶה, רֹחַב הָרְחוֹב, הַמִּדְרָכוֹת הָאֲפֹרוֹת, חֲסִיד הֶעָתִיד בְּטַלִּית מַפִּיל תַּחֲנוּנָיו לְהִתְמַלֵּא. אֵין אִישׁ עַל הָאֶבֶן, אִישׁ לֹא יוֹשֵׁב בְּצַד הַדֶּרֶךְ לְצַיֵּר אוֹ לִכְתֹּב. כָּאן הַשְּׁבִילִים עֲמוּסִים מֵתִים, צוֹעֲדִים צְפוּפִים בֵּינֵיהֶם, בּוֹדְדִים לָעַיִן — הַחַיִּים. דֶּרֶך בֵּית לֶחֶם נִמְתַּח אֶל נֶצַח הֶעָבָר, דְּרָכִים עִם מַעֲקַשִּׁים, סִפּוּרִים קוֹדְמִים מֵהֶם עֲשׂוּיִים הָעֲנָנִים, טָסִים וּמִתְכַּהִים גַּעְגּוּעִים. גַּם לְאֵלּוּ שֶׁעוֹד חַיִּים, עִוְּרִים. מַשֶּׁהוּ אָבוּד מְיַלֵּל מֵעֹמֶק הֶחָזֶה כְּתָמִיד. נִכְנָסִים אֶל הַזְּמַנִּים שֶׁלֹּא בְּתוֹרָם אֲנָשִׁים שֶׁהוֹפְכִים אֶת הָעִיר הַזּוֹ לִכְפָר. בַּשְּׂדֵרוֹת הָרְחָבוֹת חֲלוֹמוֹת, תֹּם הַחֲלוֹמוֹת, תֹּם הַחֲלִימָה, הַמַּגָּעִים הַמְּדֻיָּקִים בְּעַצְמֵנוּ, לֹא בַּאֲחֵרִים. הַתְּשׁוּקָה לִשְׂרֹט חָזָק, כְּלוֹמַר לִנְשֹׁם עָמֹק, בְּעֵץ הַתּוּת הֶעָנָק, בַּיּוֹנְקִים מִמְּנוּ חַמְצָן.

בְּשָׁעָה שָׁלוֹשׁ וָרֶבַע הַשָּׁעוֹן מְצַלְצֵל, אֲנַחְנוּ מְטַפְּסוֹת רְתוּמוֹת מְחַכּוֹת עַל גֶּדֶר הַבַּיִת הַיָּשָׁן שֶׁל אֱלוּל בְּפִנַּת הָרְחוֹב, לַהוֹרִים. מַחְנוֹת אֶת הַיַּלְקוּטִים עַל הַגָּדֵר וְנִשְׁעֲנוֹת עֲיֵפוֹת, שִׁעוּרֵי הַבַּיִת וְהַמַּחְבָּרוֹת נִלְעָסִים מְאוּסִים, מֵעֵבֶר לַגָּדֵר שְׁבִיל הָאֲבָנִים וּבַעֲבָרָיו שָׂדֶה זָעִיר, סַבְיוֹנִים וּפֵאוֹת חַמְצִיצִים, בְּאָדָר נִתַּז בֹּשֶׂם קְלִפַּת הָעֵץ, הַתִּלְתָּנִים, אַט הָאַף שָׁב לִרְאוֹת. בְּאֶגְרוֹף שֶׁנִּשְׁלַח לִמְשֹׁךְ בְּגֶזֶר אֲנַחְנוּ תּוֹלְשׁוֹת בִּשְׁעוֹת הַהֶעָזָה שִׁכְחָה, מְבָרְכוֹת בּוֹרֵא פְּרִי הָאֲדָמָה, נוֹשְׁכוֹת מוֹצְצוֹת וְנוֹשְׁכוֹת לִשְׁתּוֹת אֶת הַחֲמִיצוּת הַיְּרֻקָּה. תַּחַת הַבְּרָכָה רְחִישַׁת הַשְּׁאֵלָה זוֹ גְּזֵלָה? הַפָּנִים מִתְעַמְּלִים, מִתְכַּוְּצִים, מִתְאוֹשְׁשִׁים מִשָּׁעוֹת שֶׁל בְּהִיָּה, מַשְׁלִיכוֹת אֶת הַיֶּרֶק רֵיק אֶל הַשָּׂדֶה בַּחֲזָרָה. עַל אוֹתָהּ מִדְרָכָה צָרָה גּוֹלְשִׁים כָּל יוֹם הַנְּעוּרִים הַבּוֹהֲקִים שֶׁל סָאלְד. הַתִּיקִים תְּלוּיִים נָמוּךְ עַל הָאַגָּן (זֶה רְחוֹבִי) בְּשֵׂעָר פָּזוּר (זֶה רְחוֹבִי) עִם מַסְטִיק בְּפֶה (זֶה רְחוֹבִי) מְדַבְּרִים בְּקוֹל (זֶה רְחוֹבִי) בָּנִים וּבָנוֹת בְּלִי שַׁרְווּלִים צְהֻבִּים בַּקְּצָווֹת, יוֹרְקִים עַל הָרִצְפָּה לֵחָה וּמִלִּים, מְרַחְרְחִים אֶת הָאֲוִיר יוֹדְעִים, יוֹדְעִים, הֵם יוֹדְעִים. אוֹתָנוּ מְעַכְּבִים בַּבּוּשָׁה וְהֵם מִתְנַצְּחִים עִם הַשֶּׁמֶשׁ וְזוֹכִים. דּוֹסִית מַסְרִיחָה הֵם זוֹרְקִים לִי אֶת הַתִּיק מֵעֵבֶר לַגָּדֵר אֵל פְּאַת הַשָּׂדֶה, עַל הָאֲדָמָה. דֶּרֶךְ הַשַּׁעַר נִכְנֶסֶת וּמְרִימָה אוֹתוֹ מַהֵר מֵהַסַּבְיוֹנִים, לִמְחוֹת לִמְחוֹת. אֲנִי שׂוֹנֵאת לְהִתְכּוֹפֵף עִם חֲצָאִית כִּי אוּלַי יֵחָשְׂפוּ אֲחוֹרֵי הַבִּרְכַּיִם שֶׁאֲנִי עַצְמִי לֹא מַכִּירָה.

חיים טוקצ'ינסקי

לִפְנֵי שָׁבוּעַ שׁוּב שַׁטְנוּ צְמוּדִים בִּגְרוֹנוֹ שֶׁל קֶרֶן הַיְּסוֹד לִפְנֵי הַהִתְרַחֲבוּת הַגְּדוֹלָה אֶל קְצוֹת הָעִיר, פֵּאוֹתֶיךָ הִתְנַפְנְפוּ עַל חַמְשַׁת הַחַשְׁמַל מַלְחִינוֹת מֵחָדָשׁ גֶּשֶׁם, גּוּפְךָ רָעַד אוֹבֶר־טוֹנִים עַזִּים, חֲרִישִׁיִּים. בְּעֵינֵי חֶסֶד כְּחֻלּוֹת הִטְבַּלְתָּ יוֹם וְלַיְלָה אֶת נְהַר הָאַסְפַלְט, אֶת סִירַת הָאוֹפַנַּיִם, אֶת הָאֲנָשִׁים, טְהוֹרִים טְהוֹרִים טְהוֹרִים. בְּחַדְרֵי הָאִמּוּן הַקְּטַנִּים הָיִיתָ רִאשׁוֹן בַּבְּקָרִים, מְשַׁלֵּחַ טְרִילִים רַעֲנַנִּים, מוֹתֵחַ יְדִידֵי כֶּסֶף מַעַל גִּבְעַת רָם אֶל עֲנָנִים קַלִּים. חַיִּים, תַּלְמִיד חָכָם בְּכוֹלֵל הַקְּלִידִים לֹא הִתְעַקַּמְתָּ, לֹא שַׁחְתָּ, לֹא פָּשַׁטְתָּ קְדֻשָּׁה, יָשַׁבְתָּ זָקוּף וּמְרֻכָּז, שָׁחוֹר וְלָבָן נוֹתָר מַלְבּוּשְׁךָ. יֶלֶד חֶיְדֶר זַךְ מְלַקֵּק לְעוֹלָם דְּבַשׁ אָלֶף בֵּית שֶׁל שָׂפָה עֶלְיוֹנָה — הַזְּמַן לֹא כִּסָּה אוֹתְךָ זֻהֲמָה, לֹא קִמֵּט אֶת עֵינֶיךָ, לֹא הוֹסִיף לְךָ חָכְמָה, רְכוּב אוֹפַנַּיִם — הַמּוּחָשׁ בְּיוֹתֵר שֶׁיָּכֹל לָשֵׂאת גּוּפְךָ הַחוֹמֵק מִכְּבִידָה, גּוּפְךָ הֲלַהֲטוּטָנִי שֶׁנִּשָּׂא בְּשִׁוּוּי מִשְׁקָל נְבוּאִי עַל חוּט שְׂעָרָה. צָפִיתִי בְּעֵינֶיךָ שֶׁחִיְּכוּ תָּמִיד מַטָּה אֶל הָרְבִיעִיָּה הַקָּאמֶרִית בַּחֲצִי הגֹּרֶן הַהֶכְרֵחִי בֵּין כְּתֵפֶיךָ, בְּאַפְּךָ שֶׁרַק הַשֶּׁמֶשׁ נִשְׁקָהּ בְּפֶרְמָטוֹת חוּמוֹת זְעִירוֹת, בְּאֶצְבְּעֹתֶיךָ: עֲשָׂרָה צַוָּארִים מְנִיעִים אֶת רָאשֵׁיהֶם בְּעֶדְנָה, וְכָל אֵלֶּה טְמוּנִים עַכְשָׁו, הֲלוּמֵי דְּמָמָה, בָּאֲדָמָה, בָּאֲדָמָה.
